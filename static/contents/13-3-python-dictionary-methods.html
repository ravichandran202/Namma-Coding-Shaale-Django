Dictionary Methods

Learn all built-in dictionary methods for efficient data manipulation

Remove All Items

.clear()Removes all items from the dictionary.

student = {"name": "Ram", "age": 22}
print("Before clear:", student)
student.clear()
print("After clear:", student)
 Output
Before clear: {'name': 'Ram', 'age': 22}
After clear: {}

Copy Values

.copy() : Returns a shallow copy of the dictionary.

student = {"name": "Ram", "age": 22}
new_dict = student.copy()
print("Original:", student)
print("Copy:", new_dict)

# Modifying copy doesn't affect original
new_dict["age"] = 25
print("\nAfter modifying copy:")
print("Original:", student)
print("Copy:", new_dict)
 Output
Original: {'name': 'Ram', 'age': 22}
Copy: {'name': 'Ram', 'age': 22}

After modifying copy:
Original: {'name': 'Ram', 'age': 22}
Copy: {'name': 'Ram', 'age': 25}

Create Dict

.fromkeys(keys, value) - Creates a new dictionary with given keys and a single default value.

# Create dictionary with default value 0
keys = ["a", "b", "c"]
new_dict = dict.fromkeys(keys, 0)
print("From keys with default 0:", new_dict)

# Create dictionary with None as default
names = ["Ram", "Sita", "Krishna"]
name_dict = dict.fromkeys(names)
print("From keys with None:", name_dict)

# Practical example: initialize scores
students = ["Alice", "Bob", "Charlie"]
scores = dict.fromkeys(students, 0)
print("Student scores:", scores)
 Output
From keys with default 0: {'a': 0, 'b': 0, 'c': 0}
From keys with None: {'Ram': None, 'Sita': None, 'Krishna': None}
Student scores: {'Alice': 0, 'Bob': 0, 'Charlie': 0}

Access Elements

.get(key, default=None) : Returns value for a key if it exists, else returns default (avoids KeyError). 

student = {"name": "Ram", "age": 22}
    
# Get existing key
print("Age:", student.get("age"))

# Get non-existing key with default
print("Marks:", student.get("marks", "N/A"))

# Get non-existing key without default
print("City:", student.get("city"))

# Compare with direct access (would cause error)
# print(student["city"])  # This would raise KeyError
 Output
Age: 22
Marks: N/A
City: None

Dictionary Views

.items() - Returns all key-value pairs as a view object (iterable).
student = {"name": "Ram", "age": 22}
items_view = student.items()
print("Items:", items_view)
print("As list:", list(items_view))

# Using in loops
for key, value in student.items():
    print(f"{key}: {value}")
 Output
Items: dict_items([('name', 'Ram'), ('age', 22)])
As list: [('name', 'Ram'), ('age', 22)]
name: Ram
age: 22
.keys() - Returns all keys as a view object.
student = {"name": "Ram", "age": 22}
keys_view = student.keys()
print("Keys:", keys_view)
print("As list:", list(keys_view))

# Check if key exists
if "name" in student.keys():
    print("Name key exists")
 Output
Keys: dict_keys(['name', 'age'])
As list: ['name', 'age']
Name key exists
.values() - Returns all values as a view object.
student = {"name": "Ram", "age": 22}
values_view = student.values()
print("Values:", values_view)
print("As list:", list(values_view))

# Check if value exists
if "Ram" in student.values():
    print("Ram exists in values")
 Output
Values: dict_values(['Ram', 22])
As list: ['Ram', 22]
Ram exists in values

Remove Elements

.pop() - Remove Specific Key
student = {"name": "Ram", "age": 22, "course": "CS"}
print("Original:", student)

# Remove and get value
age = student.pop("age")
print(f"Removed age: {age}")
print("After pop:", student)

# Remove with default (no error)
marks = student.pop("marks", "Not found")
print(f"Marks: {marks}")
print("Final:", student)
 Output
Original: {'name': 'Ram', 'age': 22, 'course': 'CS'}
Removed age: 22
After pop: {'name': 'Ram', 'course': 'CS'}
Marks: Not found
Final: {'name': 'Ram', 'course': 'CS'}
popitem() - Remove Last Item
student = {"name": "Ram", "age": 22, "course": "CS"}
print("Original:", student)

# Remove last inserted item
last_item = student.popitem()
print(f"Removed: {last_item}")
print("After popitem:", student)

# Remove another
another_item = student.popitem()
print(f"Removed: {another_item}")
print("Final:", student)
 Output
Original: {'name': 'Ram', 'age': 22, 'course': 'CS'}
Removed: ('course', 'CS')
After popitem: {'name': 'Ram', 'age': 22}
Removed: ('age', 22)
Final: {'name': 'Ram'}

Get or Set Value

.setdefault() - Returns the value of the key. If the key does not exist, insert key with the default value 

student = {"name": "Ram"}
print("Original:", student)

# Key exists - returns value
name = student.setdefault("name", "Unknown")
print(f"Name: {name}")

# Key doesn't exist - adds with default
age = student.setdefault("age", 22)
print(f"Age: {age}")
print("After setdefault:", student)

# Another example
city = student.setdefault("city", "Bangalore")
print(f"City: {city}")
print("Final:", student)
 Output
Original: {'name': 'Ram'}
Name: Ram
Age: 22
After setdefault: {'name': 'Ram', 'age': 22}
City: Bangalore
Final: {'name': 'Ram', 'age': 22, 'city': 'Bangalore'}

Merge Dictionaries

.update() : Updates the dictionary with another dictionary (or key-value pairs).

student = {"name": "Ram"}
print("Original:", student)

# Update with another dictionary
student.update({"age": 22, "course": "CS"})
print("After update:", student)

# Update with keyword arguments
student.update(city="Bangalore", grade="A")
print("After keyword update:", student)

# Update with list of tuples
student.update([("country", "India"), ("phone", "1234567890")])
print("After tuple update:", student)
 Output
Original: {'name': 'Ram'}
After update: {'name': 'Ram', 'age': 22, 'course': 'CS'}
After keyword update: {'name': 'Ram', 'age': 22, 'course': 'CS', 'city': 'Bangalore', 'grade': 'A'}
After tuple update: {'name': 'Ram', 'age': 22, 'course': 'CS', 'city': 'Bangalore', 'grade': 'A', 'country': 'India', 'phone': '1234567890'}
Dictionary Methods Summary
Method Syntax Description
clear() dict.clear() Remove all items
copy() dict.copy() Create shallow copy
fromkeys() dict.fromkeys(keys, value) Create dict with default values
get() dict.get(key, default) Safe value access
items() dict.items() All key-value pairs
keys() dict.keys() All keys
values() dict.values() All values
pop() dict.pop(key, default) Remove specific key
popitem() dict.popitem() Remove last item
setdefault() dict.setdefault(key, default) Get or set value
update() dict.update(other) Merge dictionaries

Try It Yourself

Given the dictionary: student = {"name": "Ram", "age": 22, "course": "CS"}

Perform these operations:

  1. Create a copy of the dictionary
  2. Use get() to safely access "grade" (return "N/A" if not found)
  3. Use setdefault() to set "city" to "Bangalore"
  4. Remove "age" using pop() and store its value
  5. Update the dictionary with {"grade": "A", "year": 3}
  6. Get all keys and convert to list
  7. Get all values and convert to list
  8. Use fromkeys() to create a new dict with subjects and default score 0
student = {"name": "Ram", "age": 22, "course": "CS"}
print("Original:", student)

# 1. Create copy
student_copy = student.copy()
print("\n1. Copy:", student_copy)

# 2. Safe access with get()
grade = student.get("grade", "N/A")
print("\n2. Grade:", grade)

# 3. Set default city
city = student.setdefault("city", "Bangalore")
print("\n3. City:", city)
print("   After setdefault:", student)

# 4. Remove age with pop()
age = student.pop("age")
print("\n4. Removed age:", age)
print("   After pop:", student)

# 5. Update with new data
student.update({"grade": "A", "year": 3})
print("\n5. After update:", student)

# 6. Get all keys as list
keys_list = list(student.keys())
print("\n6. Keys list:", keys_list)

# 7. Get all values as list
values_list = list(student.values())
print("7. Values list:", values_list)

# 8. Create new dict with fromkeys()
subjects = ["Math", "Science", "English"]
scores = dict.fromkeys(subjects, 0)
print("\n8. Subjects with default scores:", scores)
 Output
Original: {'name': 'Ram', 'age': 22, 'course': 'CS'}

1. Copy: {'name': 'Ram', 'age': 22, 'course': 'CS'}

2. Grade: N/A

3. City: Bangalore
   After setdefault: {'name': 'Ram', 'age': 22, 'course': 'CS', 'city': 'Bangalore'}

4. Removed age: 22
   After pop: {'name': 'Ram', 'course': 'CS', 'city': 'Bangalore'}

5. After update: {'name': 'Ram', 'course': 'CS', 'city': 'Bangalore', 'grade': 'A', 'year': 3}

6. Keys list: ['name', 'course', 'city', 'grade', 'year']
7. Values list: ['Ram', 'CS', 'Bangalore', 'A', 3]

8. Subjects with default scores: {'Math': 0, 'Science': 0, 'English': 0}
